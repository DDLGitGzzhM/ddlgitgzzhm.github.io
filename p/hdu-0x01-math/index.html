    "fmt"

func gcd(a,b int) int {
    if a > b {
        a,b = b,a
    }    
    for ; a != 0 ;  {
        a, b = b%a,a
    }
    return b 
}

    var n int 
    fmt.Scan(&n)
    for i := 0 ; i < n ; i ++ {
        var a,b int 
        fmt.Scan(&a,&b)
        fmt.Println(gcd(a,b))
    }
}

    int temp ; 
    if(a > b) {
        temp = a;
        a = b ;
        b = temp;
    }

        temp = a;
        a = b%a;
        b = temp ;
    }

}
