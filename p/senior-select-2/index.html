    university,
    avg_question_cnt,
    avg_answer_cnt

    (
        SELECT
            university,
            AVG(question_cnt) AS avg_question_cnt,
            AVG(answer_cnt) AS avg_answer_cnt
        FROM
            user_profile
        GROUP BY
            university
    ) AS temp
WHERE
    avg_question_cnt < 5
    OR avg_answer_cnt < 20;
